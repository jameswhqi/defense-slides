\startMPdefinitions
vardef getEllipse =
  save n; n := 40;
  for i = 0 upto n-1:
    hide(
      theta := i * 360 / n;
      x_ := cosd theta * sqrt(1 + rho);
      y_ := sind theta * sqrt(1 - rho);
      x := mu1 + (x_ - y_) / sqrt 2 * sigma1;
      y := mu2 + (x_ + y_) / sqrt 2 * sigma2;
    )
    ((x,y) transformed t)..
  endfor cycle
enddef;
\stopMPdefinitions
